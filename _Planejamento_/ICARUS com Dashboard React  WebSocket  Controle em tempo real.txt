🔷 2. BACKEND — WEBSOCKET SERVER
📁 api/server.py
Python
from fastapi import FastAPI, WebSocket
from core.orchestrator import Orchestrator
from memory.memory_manager import MemoryManager

app = FastAPI()

orchestrator = Orchestrator()
memory = MemoryManager()

clients = []

@app.websocket("/ws")
async def websocket_endpoint(websocket: WebSocket):
    await websocket.accept()
    clients.append(websocket)

    try:
        while True:
            data = await websocket.receive_json()
            command = data.get("command")

            result = orchestrator.route(command)

            log = {
                "command": command,
                "result": result
            }

            memory.log(log)

            # broadcast
            for client in clients:
                await client.send_json(log)

    except:
        clients.remove(websocket)
🔷 3. STATUS EM TEMPO REAL (AGENTES)
📁 core/orchestrator.py (upgrade)
Python
from agents.financial_agent import FinancialAgent
from agents.dev_agent import DevAgent

class Orchestrator:

    def __init__(self):
        self.agents = [
            FinancialAgent(),
            DevAgent()
        ]

    def get_status(self):
        return [
            {
                "name": agent.name,
                "sector": agent.sector,
                "status": "idle"
            }
            for agent in self.agents
        ]

    def route(self, command):
        command = command.lower()

        for agent in self.agents:
            if agent.sector in command:
                return agent.execute(command)

        return "❌ Nenhum agente disponível"
🔷 4. FRONTEND — REACT (DASHBOARD JARVIS)
📁 ui/dashboard.jsx
JSX
import React, { useEffect, useState } from "react";

export default function Dashboard() {
  const [logs, setLogs] = useState([]);
  const [command, setCommand] = useState("");
  const [ws, setWs] = useState(null);

  useEffect(() => {
    const socket = new WebSocket("ws://localhost:8000/ws");

    socket.onmessage = (event) => {
      const data = JSON.parse(event.data);
      setLogs((prev) => [data, ...prev]);
    };

    setWs(socket);

    return () => socket.close();
  }, []);

  const sendCommand = () => {
    ws.send(JSON.stringify({ command }));
    setCommand("");
  };

  return (
    <div style={{ padding: 20, fontFamily: "monospace" }}>
      <h1>🧠 ICARUS Dashboard</h1>

      <div>
        <input
          value={command}
          onChange={(e) => setCommand(e.target.value)}
          placeholder="Digite comando..."
          style={{ width: "70%" }}
        />
        <button onClick={sendCommand}>Enviar</button>
      </div>

      <h2>📡 Logs em Tempo Real</h2>

      <div style={{ maxHeight: 400, overflow: "auto" }}>
        {logs.map((log, index) => (
          <div key={index}>
            <strong>🎙️ {log.command}</strong>
            <p>➡️ {log.result}</p>
            <hr />
          </div>
        ))}
      </div>
    </div>
  );
}
