import sqlite3

class Memory:

    def __init__(self):
        self.conn = sqlite3.connect("/opt/cfdm/database/db.sqlite")
        self.create()

    def create(self):
        self.conn.execute("""
        CREATE TABLE IF NOT EXISTS memory (
            id INTEGER PRIMARY KEY,
            event TEXT
        )
        """)

    def save(self, event):
        self.conn.execute("INSERT INTO memory (event) VALUES (?)", (event,))
        self.conn.commit()

    def recall(self):
        return self.conn.execute("SELECT * FROM memory").fetchall()

import { useState } from "react";

export default function App() {
  const [task, setTask] = useState("");

  const sendTask = async () => {
    await fetch("http://localhost:8000/task", {
      method: "POST",
      body: JSON.stringify({ task }),
    });
  };

  return (
    <div>
      <h1>CFDM Nexus</h1>
      <input onChange={(e) => setTask(e.target.value)} />
      <button onClick={sendTask}>Executar</button>
    </div>
  );
}
