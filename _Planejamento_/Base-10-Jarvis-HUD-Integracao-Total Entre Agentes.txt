class IcaroBrain:

    def __init__(self, memory, orchestrator):
        self.memory = memory
        self.orchestrator = orchestrator

    def process_input(self, user_input):
        context = self.memory.retrieve(user_input)

        decision = self.think(user_input, context)

        action = self.orchestrator.execute(decision)

        self.memory.store(user_input, action)

        return action

🎛️ ORQUESTRADOR (orchestrator.py)
Python
class Orchestrator:

    def __init__(self):
        self.agents = {}

    def register_agent(self, name, agent):
        self.agents[name] = agent

    def execute(self, decision):
        # lógica de delegação
        agent = self.agents.get("default")
        return agent.run(decision)
