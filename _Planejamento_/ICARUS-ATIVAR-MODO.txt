import json

class IcarusCore:

    def __init__(self, config_path="commands.json"):
        with open(config_path, "r", encoding="utf-8") as f:
            self.config = json.load(f)

    # -----------------------------
    # 🎯 INTERPRETADOR DE COMANDO
    # -----------------------------
    def interpret(self, command_text):
        command_text = command_text.lower()

        if "ativar modo" in command_text:
            return self.activate_mode(command_text)

        elif "convocar" in command_text:
            return self.assign_agent(command_text)

        elif "encontrar agente" in command_text:
            return self.find_agent(command_text)

        elif "montar equipe" in command_text:
            return self.build_team(command_text)

        else:
            return "❌ Comando não reconhecido"

    # -----------------------------
    # 🧠 MODOS
    # -----------------------------
    def activate_mode(self, text):
        for mode in self.config["modes"]:
            if mode.lower() in text:
                return f"✅ Modo {mode} ativado"

        return "❌ Modo não encontrado"

    # -----------------------------
    # 👤 ATRIBUIR AGENTE
    # -----------------------------
    def assign_agent(self, text):
        for agent in self.config["agents"]:
            if agent["setor"] in text:
                return f"👤 Agente {agent['nome']} convocado para tarefa"

        return "❌ Nenhum agente encontrado"

    # -----------------------------
    # 🔍 BUSCAR AGENTE
    # -----------------------------
    def find_agent(self, text):
        resultados = []

        for agent in self.config["agents"]:
            if agent["nivel"] in text or agent["setor"] in text:
                resultados.append(agent["nome"])

        if resultados:
            return f"🔎 Agentes encontrados: {', '.join(resultados)}"

        return "❌ Nenhum agente compatível"

    # -----------------------------
    # 👥 MONTAR EQUIPE
    # -----------------------------
    def build_team(self, text):
        equipe = []

        for agent in self.config["agents"]:
            if agent["disponibilidade"] == "alta":
                equipe.append(agent["nome"])

        return f"🚀 Equipe montada: {', '.join(equipe)}"


# -----------------------------
# 🚀 EXECUÇÃO
# -----------------------------

import json
import os

class MemoryManager:
    def __init__(self, path="memory/memory.json"):
        self.path = path
        if not os.path.exists(self.path):
            self._init_memory()

    def _init_memory(self):
        with open(self.path, "w") as f:
            json.dump({"logs": [], "state": {}}, f)

    def read(self):
        with open(self.path, "r") as f:
            return json.load(f)

    def write(self, data):
        with open(self.path, "w") as f:
            json.dump(data, f, indent=2)

    def log(self, entry):
        data = self.read()
        data["logs"].append(entry)
        self.write(data)

from agents.financial_agent import FinancialAgent
from agents.dev_agent import DevAgent

class Orchestrator:

    def __init__(self):
        self.agents = [
            FinancialAgent(),
            DevAgent()
        ]

    def route(self, command):
        command = command.lower()

        for agent in self.agents:
            if agent.sector in command:
                return agent.execute(command)

        return "❌ Nenhum agente disponível para essa tarefa"

from fastapi import FastAPI
from core.orchestrator import Orchestrator
from memory.memory_manager import MemoryManager

app = FastAPI()

orchestrator = Orchestrator()
memory = MemoryManager()

@app.post("/command")
def execute_command(data: dict):
    command = data.get("command")

    result = orchestrator.route(command)

    memory.log({
        "command": command,
        "result": result
    })

    return {
        "status": "ok",
        "result": result
    }

@app.get("/memory")
def get_memory():
    return memory.read()
