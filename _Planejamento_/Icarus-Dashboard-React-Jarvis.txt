🔷 2. COMPONENTE PRINCIPAL (REACT)

JSX
import React, { useEffect, useState } from "react";
import { motion } from "framer-motion";

const Panel = ({ title, children }) => (
  <motion.div
    initial={{ opacity: 0, scale: 0.95 }}
    animate={{ opacity: 1, scale: 1 }}
    style={{
      background: "rgba(0,0,0,0.6)",
      border: "1px solid cyan",
      borderRadius: "12px",
      padding: "10px",
      color: "cyan",
      backdropFilter: "blur(10px)"
    }}
  >
    <h3>{title}</h3>
    {children}
  </motion.div>
);

export default function IcarusOS() {
  const [logs, setLogs] = useState([]);
  const [agents, setAgents] = useState([
    { name: "Financeiro", status: "idle" },
    { name: "Dev", status: "idle" }
  ]);
  const [command, setCommand] = useState("");
  const [ws, setWs] = useState(null);

  useEffect(() => {
    const socket = new WebSocket("ws://localhost:8000/ws");

    socket.onmessage = (event) => {
      const data = JSON.parse(event.data);

      setLogs((prev) => [data, ...prev]);

      // Atualiza status fake (expandir depois)
      setAgents((prev) =>
        prev.map((a) =>
          data.command.includes(a.name.toLowerCase())
            ? { ...a, status: "working" }
            : a
        )
      );
    };

    setWs(socket);
    return () => socket.close();
  }, []);

  const sendCommand = () => {
    ws.send(JSON.stringify({ command }));
    setCommand("");
  };

  return (
    <div
      style={{
        display: "grid",
        gridTemplateColumns: "1fr 1fr",
        gridTemplateRows: "1fr 1fr",
        gap: "10px",
        height: "100vh",
        padding: "10px",
        background: "black"
      }}
    >
      {/* AGENT MAP */}
      <Panel title="🧠 AGENT MAP">
        {agents.map((agent, i) => (
          <motion.div
            key={i}
            animate={{
              scale: agent.status === "working" ? 1.1 : 1
            }}
            style={{
              margin: "5px",
              padding: "10px",
              border: "1px solid cyan",
              borderRadius: "8px"
            }}
          >
            {agent.name} — {agent.status}
          </motion.div>
        ))}
      </Panel>

      {/* LOGS */}
      <Panel title="📡 LIVE LOGS">
        <div style={{ maxHeight: 300, overflow: "auto" }}>
          {logs.map((log, i) => (
            <div key={i}>
              🎙️ {log.command}
              <br />
              ➡️ {log.result}
              <hr />
            </div>
          ))}
        </div>
      </Panel>

      {/* COMMAND */}
      <Panel title="🎛️ COMMAND HUB">
        <input
          value={command}
          onChange={(e) => setCommand(e.target.value)}
          style={{ width: "80%" }}
        />
        <button onClick={sendCommand}>Enviar</button>
      </Panel>

      {/* STATUS */}
      <Panel title="📊 SYSTEM STATUS">
        <p>CPU: Simulado</p>
        <p>Agentes Ativos: {agents.length}</p>
        <p>Status: Online</p>
      </Panel>
    </div>
  );
}
