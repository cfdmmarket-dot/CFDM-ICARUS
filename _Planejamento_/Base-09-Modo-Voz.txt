🤖 PASSO 3 — AGENTES AUTÔNOMOS “VIVOS”
🔁 LOOP AUTÔNOMO

import time

class AutonomousAgent:

    def __init__(self, name, ai):
        self.name = name
        self.ai = ai

    def think(self):
        print(f"[{self.name}] Pensando...")
        self.ai.process("Gerar oportunidade de lucro")

    def act(self):
        print(f"[{self.name}] Executando ação")

    def live(self):
        while True:
            self.think()
            self.act()
            time.sleep(10)
