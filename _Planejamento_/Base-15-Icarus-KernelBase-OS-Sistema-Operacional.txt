⚡ 1. EVENT BUS (O CORAÇÃO)
Python
# event_bus.py

class EventBus:
    def __init__(self):
        self.listeners = {}

    def subscribe(self, event_type, callback):
        if event_type not in self.listeners:
            self.listeners[event_type] = []
        self.listeners[event_type].append(callback)

    def publish(self, event_type, data=None):
        if event_type in self.listeners:
            for callback in self.listeners[event_type]:
                callback(data)

class ModuleManager:
    def __init__(self):
        self.modules = []

    def register(self, module):
        self.modules.append(module)
        module.on_load()

    def get_modules(self):
        return self.modules

⚡ 4. EXECUTOR MODULE (AÇÃO REAL)

# modules/executor_module.py

import os
from modules.base_module import BaseModule

class ExecutorModule(BaseModule):

    def on_load(self):
        self.event_bus.subscribe("execute_command", self.run)

    def run(self, data):
        command = data.get("command")
        print(f"[EXECUTOR] Rodando: {command}")

        try:
            result = os.popen(command).read()
            print(result)
        except Exception as e:
            print(e)

from modules.base_module import BaseModule

class MemoryModule(BaseModule):

    def __init__(self, name, event_bus):
        super().__init__(name, event_bus)
        self.memory = []

    def on_load(self):
        self.event_bus.subscribe("memory.store", self.store)

    def store(self, data):
        self.memory.append(data)
        print("[MEMORY] Dado armazenado")

import time

class Scheduler:

    def __init__(self, event_bus):
        self.event_bus = event_bus

    def run(self):
        while True:
            # exemplo de evento automático
            self.event_bus.publish("execute_command", {

# main.py

from kernel import IcarusKernel
from modules.executor_module import ExecutorModule
from modules.memory_module import MemoryModule
from scheduler import Scheduler

def main():
    kernel = IcarusKernel()

    # carregar módulos
    executor = ExecutorModule("executor", kernel.event_bus)
    memory = MemoryModule("memory", kernel.event_bus)

    kernel.load_module(executor)
    kernel.load_module(memory)

    kernel.start()

    # iniciar scheduler (autonomia)
    scheduler = Scheduler(kernel.event_bus)
    scheduler.run()

if __name__ == "__main__":
    main()
