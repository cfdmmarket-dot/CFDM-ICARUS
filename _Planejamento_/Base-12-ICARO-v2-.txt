🎯 goal_manager.py

class GoalManager:

    def __init__(self):
        self.goals = []

    def add_goal(self, goal):
        self.goals.append({
            "goal": goal,
            "status": "pending"
        })

    def get_next_goal(self):
        for g in self.goals:
            if g["status"] == "pending":
                return g
        return None

🔁 loop.py (O CORAÇÃO)
Python
import time

class AutonomousLoop:

    def __init__(self, goal_manager, planner, orchestrator):
        self.goal_manager = goal_manager
        self.planner = planner
        self.orchestrator = orchestrator

    def run(self):
        print("⚡ ÍCARO AUTÔNOMO ATIVO")

        while True:
            goal = self.goal_manager.get_next_goal()

            if goal:
                print(f"🎯 Novo objetivo: {goal['goal']}")

                plan = self.planner.create_plan(goal["goal"])

                for step in plan:
                    print(f"➡️ Executando: {step}")
                    self.orchestrator.execute({
                        "action": "execute_command",
                        "command": step
                    })

                goal["status"] = "done"

            time.sleep(5)
